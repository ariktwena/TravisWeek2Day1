//Callback example
var cb = function (n1, n2, callback) {
    return "Result from the two numbers: " + n1 + "+" + n2 + "=" + callback(n1, n2);
};

console.log("Callback");
console.log(add(1, 2)); //Den ligger de to tal sammen
console.log(add); //Den kalder add funktionen, men der er ikke variabler i den, så den udfører intet
console.log(add(1, 2, 3)); //Der mangler et tal i add funktionen, så den ser bort fra den sidste variable
console.log(add(1)); //Den mangler en variable for at udføre funktionen og derfor får vi NaN
console.log(cb(3, 3, add)); //Vi giver "add" funktionen med som callback, som bliver afviklet i funktionen "cb": Result from the two numbers: 3 + 3 = 6
console.log(cb(4, 3, sub)); //Vi giver "sub" funktionen med som callback, som bliver afviklet i funktionen "cb": Result from the two numbers: 4 - 3 = 1
//console.log(cb(3,3,add())); //Callback er ikke en funktion da den ikke opfylder funktionskriterierne
console.log(cb(3, "hh", add)); //Den ændre datatypen til String da "hh" er string og sææter det sammen med 3:Result from the two numbers: 3 + hh = 3hh

console.log("Module Pattern");
function EmployeeDetails() {
    let nameEmployee = "Mayank";
    let age = 30;
    let designation = "Developer";
    const setName = function (nameEmployee) {
        this.nameEmployee = nameEmployee;
    };
    const setAge = function (age) {
        this.age = age;
    };

    const getInfo = function (nameEmployee, age) {
        this.nameEmployee = nameEmployee;
        this.age = age;

        return `${this.nameEmployee}, ${this.age}`;
    };

    const getInfo2 = function (newEmployee) {
        console.log(`${newEmployee.nameEmployee}, ${newEmployee.age}`);
    };

    return {
        nameEmployee: nameEmployee,
        age: age,
        designation: designation,
        setName: setName,
        setAge: setAge,
        getInfo: getInfo,
        getInfo2: getInfo2
    };
}
;

var newEmployee = EmployeeDetails();
console.log(newEmployee);
console.log(newEmployee.nameEmployee);
console.log(newEmployee.age);
newEmployee.setName("John");
newEmployee.setAge(98);
console.log(newEmployee);
console.log(newEmployee.nameEmployee);
console.log(newEmployee.age);
console.log(newEmployee.getInfo(newEmployee.nameEmployee, newEmployee.age));
newEmployee.getInfo2(newEmployee);
